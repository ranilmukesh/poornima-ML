if PreRdiaage is None or postblage is None:
return None

try:
x = float(PreRdiaage)
age = float(postblage)
except (ValueError, TypeError):
return None

if x <= 0 or age <= 0:
return None

# Case 1: Year of diagnosis
if 1900 <= x <= current_year:
return current_year - x

# Case 2: Likely duration (small value)
if x < 20:
return x

# Case 3: Valid onset age
if x <= age:
return age - x

# Case 4: Possible swapped ages
if x > age and x < 120 and age < 120:
return x - age # assume swap

def get_category(v):

return "Normoglycemia"
elif v <= 6.4:
return "Prediabetes"
else:
return "Diabetes"

post_cat = get_category(PostBLHBA1C)

# ---------- Step 2: Trajectory ----------
order = {"Normoglycemia": 0, "Prediabetes": 1, "Diabetes": 2}

if order[post_cat] < order[pre_cat]:
traj = "Regression"
elif order[post_cat] == order[pre_cat]:
traj = "Persistence"
else:
traj = "Progression"

outcome_line = f"Predicted outcome: {pre_cat} → {post_cat} ({traj})"

# ---------- Step 3: Delta ----------
delta = PreBLHBA1C - PostBLHBA1C # positive = improvement

# Defaults (UI-safe)
response_line = ""
target_line = ""

# ---------- Step 4: Response classification (baseline Diabetes only) ----------
if pre_cat == "Diabetes":
if delta >= 1.0:
response_line = f"Predicted response: Major improvement – Risk reduction achieved (ΔHbA1c {delta:+.2f}%)"
elif delta >= 0.5:
response_line = f"Predicted response: Clinically meaningful improvement (ΔHbA1c {delta:+.2f}%)"

response_line = f"Predicted response: Non-response (ΔHbA1c {delta:+.2f}%)"
